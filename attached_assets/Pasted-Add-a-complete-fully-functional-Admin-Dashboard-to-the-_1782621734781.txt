MIDDLEWARE

Create middleware.ts at root:

import { getToken } from "next-auth/jwt"
import { NextResponse } from "next/server"

export async function middleware(req) {
  const token = await getToken({ req })
  const isAdminRoute = req.nextUrl.pathname.startsWith("/admin")

  if (isAdminRoute) {
    if (!token) return NextResponse.redirect(new URL("/login", req.url))
    if (token.role !== "ADMIN") return NextResponse.redirect(new URL("/", req.url))
  }
  return NextResponse.next()
}

export const config = { matcher: ["/admin/:path*"] }

Make sure the JWT token includes the user role — update the NextAuth callbacks in /api/auth/[...nextauth] to include role in both jwt() and session() callbacks.
