Build a complete, production-ready Arabic e-commerce web application called "كنوز" (Knouz) for the Egyptian market. This is a full-stack project. Do not use placeholders or TODOs — generate all code fully working from the start.

---

TECH STACK

- Frontend: Next.js 14 (App Router), TypeScript, Tailwind CSS
- Database: PostgreSQL using Prisma ORM (use Replit's built-in PostgreSQL)
- Auth: NextAuth.js with credentials provider (email + password)
- Image storage: Cloudinary (use environment variables for keys)
- Payment: Paymob API (Egyptian payment gateway)
- Email: Nodemailer with Gmail SMTP
- State: Zustand for cart and wishlist
- Validation: Zod for all API inputs

DATABASE — Create this full Prisma schema

model User {
  id        String   @id @default(cuid())
  name      String
  email     String   @unique
  password  String
  phone     String?
  role      Role     @default(CUSTOMER)
  orders    Order[]
  reviews   Review[]
  wishlist  Wishlist[]
  addresses Address[]
  createdAt DateTime @default(now())
}

enum Role { CUSTOMER ADMIN }

model Category {
  id       String    @id @default(cuid())
  nameAr   String
  slug     String    @unique
  image    String?
  products Product[]
}

model Product {
  id            String      @id @default(cuid())
  nameAr        String
  descriptionAr String
  price         Float
  comparePrice  Float?
  stock         Int         @default(0)
  images        String[]
  categoryId    String
  category      Category    @relation(fields: [categoryId], references: [id])
  featured      Boolean     @default(false)
  reviews       Review[]
  orderItems    OrderItem[]
  wishlist      Wishlist[]
  createdAt     DateTime    @default(now())
}

model Order {
  id              String      @id @default(cuid())
  userId          String?
  user            User?       @relation(fields: [userId], references: [id])
  status          OrderStatus @default(PENDING)
  total           Float
  shippingAddress Json
  paymentMethod   String
  paymentStatus   String      @default("unpaid")
  couponCode      String?
  discount        Float       @default(0)
  items           OrderItem[]
  createdAt       DateTime    @default(now())
}

enum OrderStatus { PENDING PROCESSING SHIPPED DELIVERED CANCELLED }

model OrderItem {
  id        String  @id @default(cuid())
  orderId   String
  order     Order   @relation(fields: [orderId], references: [id])
  productId String
  product   Product @relation(fields: [productId], references: [id])
  quantity  Int
  price     Float
}

model Review {
  id        String   @id @default(cuid())
  userId    String
  user      User     @relation(fields: [userId], references: [id])
  productId String
  product   Product  @relation(fields: [productId], references: [id])
  rating    Int
  comment   String?
  createdAt DateTime @default(now())
}

model Coupon {
  id         String      @id @default(cuid())
  code       String      @unique
  discount   Float
  type       CouponType
  minOrder   Float       @default(0)
  expiresAt  DateTime?
  usageLimit Int?
  usedCount  Int         @default(0)
}

enum CouponType { PERCENTAGE FIXED }

model Wishlist {
  id        String  @id @default(cuid())
  userId    String
  user      User    @relation(fields: [userId], references: [id])
  productId String
  product   Product @relation(fields: [productId], references: [id])
}

model Address {
  id          String  @id @default(cuid())
  userId      String
  user        User    @relation(fields: [userId], references: [id])
  name        String
  phone       String
  governorate String
  city        String
  street      String
  isDefault   Boolean @default(false)
}

API ROUTES — Build all of these

POST   /api/auth/[...nextauth]     — NextAuth handler
GET    /api/products               — List products (with filters: category, minPrice, maxPrice, sort, page)
POST   /api/products               — Create product (admin only)
PUT    /api/products/[id]          — Update product (admin only)
DELETE /api/products/[id]          — Delete product (admin only)
GET    /api/products/[slug]        — Single product by slug
GET    /api/categories             — List categories
POST   /api/orders                 — Create order, send confirmation email
GET    /api/orders                 — List orders (admin) or user orders (customer)
PUT    /api/orders/[id]            — Update order status (admin)
GET    /api/orders/track           — Track order by orderNumber + phone
POST   /api/coupons/validate       — Validate coupon code, return discount
POST   /api/upload                 — Upload image to Cloudinary, return URL
GET    /api/admin/stats            — Dashboard stats (admin only)
